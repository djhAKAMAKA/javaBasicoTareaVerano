package org.tarea.controller;

import org.tarea.model.Pedido;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class PedidoController {
    private List<Pedido> pedidos;

    public PedidoController() {
        this.pedidos = new ArrayList<>();
        pedidos.add(new Pedido("001", LocalDate.of(2023, 8, 10), "Juan Cuesta", "Ordenador", 2, 1200.50));
        pedidos.add(new Pedido("002", LocalDate.of(2023, 8, 15), "Luisma Garcia", "Telefono", 1, 800.00));
        pedidos.add(new Pedido("003", LocalDate.of(2023, 8, 20), "Antonio Recio", "Tablet", 2, 200.50));
        pedidos.add(new Pedido("004", LocalDate.of(2023, 8, 22), "Raquel Villanueva", "Reloj", 2, 250.00));
        pedidos.add(new Pedido("005", LocalDate.of(2023, 8, 25), "Macu Colmenero", "Cámara", 1, 350.00));
    }

    public void crearPedido() {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Ingrese código del pedido: ");
        String codigo = scanner.nextLine();

        System.out.print("Ingrese fecha del pedido (YYYY-MM-DD): ");
        LocalDate fechaPedido = LocalDate.parse(scanner.nextLine());

        System.out.print("Ingrese nombre del cliente: ");
        String nombreCliente = scanner.nextLine();

        System.out.print("Ingrese nombre del producto: ");
        String producto = scanner.nextLine();

        System.out.print("Ingrese cantidad de unidades: ");
        int unidades = Integer.parseInt(scanner.nextLine());

        System.out.print("Ingrese precio del producto: ");
        double precio = Double.parseDouble(scanner.nextLine());

        Pedido nuevoPedido = new Pedido(codigo, fechaPedido, nombreCliente, producto, unidades, precio);
        pedidos.add(nuevoPedido);
        System.out.println("Pedido creado con éxito: " + nuevoPedido);
    }

    public void modificarPedido() {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Ingrese el código del pedido a modificar: ");
        String codigo = scanner.nextLine();

        Pedido pedido = buscarPedidoPorCodigo(codigo);
        if (pedido != null) {
            System.out.print("Ingrese nueva fecha del pedido (YYYY-MM-DD): ");
            LocalDate nuevaFecha = LocalDate.parse(scanner.nextLine());
            pedido.setFechaPedido(nuevaFecha);

            System.out.print("Ingrese nuevas unidades: ");
            int nuevasUnidades = Integer.parseInt(scanner.nextLine());
            pedido.setUnidades(nuevasUnidades);

            System.out.println("Pedido modificado con éxito: " + pedido);
        } else {
            System.out.println("Pedido no encontrado.");
        }
    }

    public void mostrarDiasDesdePedido() {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Ingrese el código del pedido: ");
        String codigo = scanner.nextLine();

        Pedido pedido = buscarPedidoPorCodigo(codigo);
        if (pedido != null) {
            long dias = pedido.calcularDiasDesdePedido();
            System.out.println("Días desde que se hizo el pedido: " + dias);
        } else {
            System.out.println("Pedido no encontrado.");
        }
    }

    public void eliminarPedido() {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Ingrese el código del pedido a eliminar: ");
        String codigo = scanner.nextLine();

        Pedido pedido = buscarPedidoPorCodigo(codigo);
        if (pedido != null) {
            pedidos.remove(pedido);
            System.out.println("Pedido eliminado con éxito.");
        } else {
            System.out.println("Pedido no encontrado.");
        }
    }

    private Pedido buscarPedidoPorCodigo(String codigo) {
        for (Pedido pedido : pedidos) {
            if (pedido.getCodigo().equals(codigo)) {
                return pedido;
            }
        }
        return null;
    }

    public void listarPedidos() {
        if (pedidos.isEmpty()) {
            System.out.println("No hay pedidos registrados.");
        } else {
            for (Pedido pedido : pedidos) {
                System.out.println(pedido);
            }
        }
    }
}
